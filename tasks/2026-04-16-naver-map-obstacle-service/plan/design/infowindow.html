SVG 구조: path(핀 외형) + circle(내부 흰 원) + circle(위험도 색상 점)
React 구현 시: dangerLevel prop에 따라 fill 색상을 동적으로 변경

SVG 마커 소스 코드 보기 
/* React 마커 컴포넌트 예시 */
const MARKER_COLORS = {
  '위험': '#FF4444',
  '주의': '#FFA500',
  '안전': '#44BB44',
};

const markerSvg = (dangerLevel) => {
  const color = MARKER_COLORS[dangerLevel] || '#2D3E50';
  return `data:image/svg+xml,${encodeURIComponent(`
    <svg width="32" height="40" viewBox="0 0 32 40" xmlns="http://www.w3.org/2000/svg">
      <path d="M16 0C7.163 0 0 7.163 0 16C0 27 16 40 16 40C16 40 32 27 32 16C32 7.163 24.837 0 16 0Z"
            fill="${color}"/>
      <circle cx="16" cy="16" r="7" fill="rgba(255,255,255,0.95)"/>
      <circle cx="16" cy="16" r="3.5" fill="${color}"/>
    </svg>
  `)}`;
};

// naver.maps.Marker icon 설정
new naver.maps.Marker({
  icon: { url: markerSvg(obstacle.dangerLevel), size: new naver.maps.Size(32, 40),
          anchor: new naver.maps.Point(16, 40) }
});
