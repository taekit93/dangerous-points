WishlistForm.jsx
interface WishlistFormProps {
  position : { lat: number; lng: number }  // 지도 클릭 좌표 (필수)
  onSave   : (data: WishlistItem) => void   // 저장 시 호출
  onCancel : () => void                     // 취소 시 호출
}

interface WishlistItem {
  name     : string                // 장소명
  category : WishlistCategory      // 카테고리
  address ?: string                // 주소 (선택)
  lat      : number                // 위도
  lng      : number                // 경도
}

type WishlistCategory =
  | '맛집' | '카페' | '관광지' | '공원'
  | '쇼핑' | '문화/예술' | '기타'
