interface WishlistListProps {
  items      : WishlistListItem[]          // 전체 위시리스트 아이템

}

interface WishlistListItem {
  id          : string
  name        : string
  category    : WishlistCategory
  address    ?: string
  lat         : number
  lng         : number
  visitCount  : number          // 0이면 미방문
  createdAt   : string          // ISO 날짜
}
