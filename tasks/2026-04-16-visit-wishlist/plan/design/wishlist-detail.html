Props 인터페이스

WishlistDetail.jsx
interface WishlistDetailProps {
  item       : WishlistDetailItem                    // 상세 데이터
  onVisit    : (id: string) => void                  // "방문 완료" 버튼 클릭 → visitCount++
  onAddReview: (id: string, review: ReviewInput) => void  // 후기 등록
  onDelete   : (id: string) => void                  // 장소 삭제
  onClose    : () => void                            // 패널 닫기
}

interface WishlistDetailItem {
  id         : string
  name       : string
  category   : WishlistCategory
  address   ?: string
  lat        : number
  lng        : number
  visitCount : number
  reviews    : Review[]
  createdAt  : string
}

interface Review {
  id        : string
  text      : string
  visitNum ?: number    // 몇 번째 방문 후기인지 (선택)
  createdAt : string
}

interface ReviewInput {
  text     : string
  visitNum?: number
}
